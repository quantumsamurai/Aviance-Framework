005package edu.qs.frc.modules;
006
007import edu.qs.frc.hardware.Hardware;
008import edu.qs.frc.threading.AvianceThread;
009import edu.qs.frc.threading.AvianceThreadInterruptedException;
010import edu.qs.frc.threading.AvianceThreadManager;
011import edu.wpi.first.wpilibj.Joystick;
012import edu.wpi.first.wpilibj.Talon;
013
014/**
015 *
016 * @author pawel
017 */
018public class AvianceTankDrive extends AvianceThread{
019    
020    /**
021     * This block automatically adds the class to its respective group when an instance is spawned
022     * ,to disable this comment out the line
023     */
024    {
025        AvianceThreadManager.getInstance().addThread(AvianceRobot.teleopThreads, new AvianceTankDrive());
026    }
027    
028    private Talon front_left = Hardware.talons[Hardware.TALON_DRIVE_FRONT_LEFT];
029    private Talon front_right = Hardware.talons[Hardware.TALON_DRIVE_FRONT_RIGHT];
030    private Talon rear_left = Hardware.talons[Hardware.TALON_DRIVE_REAR_LEFT];
031    private Talon rear_right = Hardware.talons[Hardware.TALON_DRIVE_REAR_RIGHT];
032    private Joystick left = Hardware.joysticks[Hardware.JOYSTICK_LEFT];
033    private Joystick right = Hardware.joysticks[Hardware.JOYSTICK_RIGHT];
034    
035    public void run(){
036        try{
037            double left_val, right_val;
038            
039            while(true){
040                left_val = left.getY();
041                right_val = right.getY();
042                
043                front_left.set(left_val);
044                front_right.set(right_val);
045                rear_left.set(left_val);
046                rear_right.set(right_val);
047                
048                this.AvianceThreadHousekeeping(sleepTime);
049            }
050        }
051        catch(AvianceThreadInterruptedException e){
052            
053        }
054    }
055    
056}
