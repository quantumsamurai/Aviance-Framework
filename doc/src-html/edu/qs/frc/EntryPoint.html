007import edu.qs.frc.modules.AvianceRobot;
008import edu.qs.frc.modules.AvianceSpacialTracking;
009import edu.qs.frc.modules.AvianceTankDrive;
010import javax.microedition.midlet.MIDlet;

012
013/**
014 * The startApp method of this class is called by the VM to start the
015 * application.
016 * 
017 * The entry point is responsible for starting and monitoring all robot subsystems
018 */
019public class EntryPoint extends MIDlet {
020
021    protected void startApp() throws MIDletStateChangeException {
022        
023        //start client connection to rhaspberry pi
024        
025        /*
026         * This line of code alone is enough to start the bot, the instancization
027         * code takes care of creating the needed thred groups and starting the
028         * one that contains the robot thread, at this point the robot is
029         * self maintained and no outside access may stop it (except for a call
030         * to interrupt(), this stops the robot)
031         * 
032         * ALL MODULES ARE RESPONSIBLE FOR ADDING THEMSELVES TO THE THREADGROUP THEY BELONG TO
033         */
034        AvianceRobot robot = new AvianceRobot();
035        AvianceTankDrive tankdrive = new AvianceTankDrive();
036        AvianceSpacialTracking spacialtracking = new AvianceSpacialTracking();
037        
038        try{
039            this.wait();
040        }
041        catch(InterruptedException e){
042            robot.interrupt();
043        }
044    }
045
046    protected void pauseApp() {
047    }
048
049    protected void destroyApp(boolean unconditional) throws MIDletStateChangeException {
050        //close connections
051    }
052    
053}
